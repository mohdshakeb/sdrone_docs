## Usage

<CodeBlock language="tsx" filename="example.tsx">
```tsx
import TimeInput from '@/components/ui/TimeInput';

// Basic usage
<TimeInput
  value={time}
  onChange={(e) => setTime(e.target.value)}
/>

// With error state
<TimeInput
  value={time}
  onChange={(e) => setTime(e.target.value)}
  hasError={!!errors.time}
/>

// Small size
<TimeInput
  size="sm"
  value={time}
  onChange={(e) => setTime(e.target.value)}
/>
```
