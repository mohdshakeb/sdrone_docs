## Usage

<CodeBlock language="tsx" filename="example.tsx">
```tsx
import FilterChip from '@/components/ui/FilterChip';

// Default state
<FilterChip>Report Type</FilterChip>

// Selected with value
<FilterChip
  selected
  value="First Aid"
  onClear={() => handleClear()}
>
  Report Type
</FilterChip>

// Multiple selected (count)
<FilterChip
  count={3}
  onClear={() => handleClear()}
>
  Project
</FilterChip>

// Disabled
<FilterChip disabled>Category</FilterChip>
```
