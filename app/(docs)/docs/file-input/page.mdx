## Usage

<CodeBlock language="tsx" filename="example.tsx">
```tsx
import FileInput from '@/components/ui/FileInput';

// Basic usage
<FileInput
  accept=".pdf,.doc,.docx"
  files={files}
  onChange={setFiles}
  placeholder="Upload documents"
/>

// Multiple files with size limit
<FileInput
  accept=".pdf,.doc,.docx"
  multiple
  maxFiles={3}
  maxSize={5242880} // 5MB
  files={files}
  onChange={setFiles}
  placeholder="Upload documents"
  helpText="Maximum 3 files, 5MB each"
/>

// With error state
<FileInput
  files={files}
  onChange={setFiles}
  hasError={!!errors.files}
  placeholder="Required file upload"
/>
```
